\begin{sageCell}
import numpy as np
import matplotlib.pyplot as plt

x = np.linspace(-10,10,100)
y = x**2-x+1
plt.plot(x,y)
plt.xlabel('$x$')
plt.ylabel('$y$')
plt.title('A graph of $x^2-x+1$')
plt.show()
\end{sageCell}
